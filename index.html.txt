  <script>
    // LOGIN FUNCTION
    function login(){
      const user=document.getElementById("username").value.trim();
      const pass=document.getElementById("password").value.trim();
      if(user==="zack" && pass==="1234"){ // contoh validasi
        document.getElementById("loginCard").classList.add("hidden");
        document.getElementById("appCard").classList.remove("hidden");
      }else{
        alert("❌ Username atau password salah!");
      }
    }

    // pilih bug/sender
    document.querySelectorAll(".bug-card, .sender-card").forEach(card=>{
      card.addEventListener("click",()=>{
        const group=card.classList.contains("bug-card")?".bug-card":".sender-card";
        document.querySelectorAll(group).forEach(c=>c.classList.remove("selected"));
        card.classList.add("selected");
      });
    });

    // drag scroll bug menu
    const bugMenu=document.getElementById("bugMenu");
    let isDown=false,startX,scrollLeft;
    bugMenu.addEventListener("mousedown",e=>{
      isDown=true; startX=e.pageX-bugMenu.offsetLeft; scrollLeft=bugMenu.scrollLeft;
    });
    bugMenu.addEventListener("mouseleave",()=>isDown=false);
    bugMenu.addEventListener("mouseup",()=>isDown=false);
    bugMenu.addEventListener("mousemove",e=>{
      if(!isDown) return;
      e.preventDefault();
      const x=e.pageX-bugMenu.offsetLeft;
      const walk=(x-startX)*2;
      bugMenu.scrollLeft=scrollLeft-walk;
    });

    // SEND BUG
    function validateAndSend(){
      const target=document.getElementById("target").value.trim();
      const bugSelected=document.querySelector(".bug-card.selected");
      const senderSelected=document.querySelector(".sender-card.selected");

      if(!target){ alert("Nomor target belum diisi!"); return; }
      if(!bugSelected){ alert("Pilih menu bug dulu!"); return; }
      if(!senderSelected){ alert("Pilih sender dulu!"); return; }

      document.getElementById("targetNumber").innerText=target;
      document.getElementById("overlay").classList.add("active");
    }
    function hideOverlay(){ document.getElementById("overlay").classList.remove("active"); }
  </script>
